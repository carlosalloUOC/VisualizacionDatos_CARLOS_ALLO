# Importamos los módulos de python que nos permiten realizar este tipo de gráficos
from matplotlib_venn import venn3, venn3_circles
from matplotlib import pyplot as plt
  
# Intorducimos los datos anteriores según el orden que nos indica la documentación para que se pinte la gráfica deseada
# Además, se escogen colores cuyas combinaciones (ya que se superponen) sean diferenciables y que no hagan daño a la vista
#como podría ocurrir si usáramos los colores primarios
venn3(subsets=(527, 373, 376, 83, 24, 149, 299), 
      set_labels=('Precio alto', 'Área grande', 'Más de 3 dormitorios'), 
      set_colors=("orange", "blue", "red"), alpha=0.7)
  
#  Creamos circulos en los bordes para ver la distinción de colores y hacerlo algo más visual
venn3_circles(subsets=(527, 373, 376, 83, 24, 149, 299),
              linestyle="dashed", linewidth=2)
  
# Introducimos título
plt.title("Venn Diagram que compara las casas con los precios más altos, área más grandes y con más de 3 dormitorios")
plt.show()
